class Solution {
    public int reverse(int x) {
        
        int res=0;
        int rem=0;
     
        while(x!=0)
        {
            rem=x%10;
            res=res*10+rem;
            if(rem!=(res%10))
            {
                return 0;
            }
               x=x/10;
        }   

        
      if(res > 2147483647 || res < -2147483648){
          return 0;
      }
    
        return res;
     
        
    }
}
